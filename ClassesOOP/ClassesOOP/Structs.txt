        // a user defined type that allows to contain multiple fields and methods
        struct Rectangle
        {
            public double length;
            public double width;

            public Rectangle(double l = 1, 
                double w = 1)
            {
                length = l;
                width = w;
            }

            public double Area()
            {
                return length * width;
            }
